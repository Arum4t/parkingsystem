package com.parkit.parkingsystem.config;


import java.io.FileInputStream;
import java.io.IOException;
import java.util.Properties;


public class ReadPropertiesFile {

    public static void readProperty(String readProperty) throws IOException {

        FileInputStream fis = new FileInputStream("src/main/resources/config.properties");

        Properties properties = new Properties();

        properties.load(fis);

    }
    public String getUsername () {
        Properties username = new Properties();
        return username.getProperty("username");
    }
    public String getPassword () {
        Properties password = new Properties();
        return password.getProperty("password");
    }
    public String getUrl1 () {
        Properties url1 = new Properties();
        return url1.getProperty("url1");
    }
    public String getUrl2 () {
        Properties url2 = new Properties();
        return url2.getProperty("url2");
    }
}
